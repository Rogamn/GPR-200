#version 300 es

// Input attributes
layout (location = 0) in vec4 aPosition;
layout (location = 1) in vec3 aNormal;

// Varyings
out vec4 vColor, vPosition;
out vec3 vNormal;

// Input uniforms
uniform mat4 uModelMatrix, uViewMatrix, uViewProjMatrix;

// sPointLight: data structure to hold info about lights
// center:    center of light
// color:     color of light
// intensity: intensity of light

struct sPointLight
{
	vec3 pos, color;
	float intensity;
};

// initLight: function to initialize a light
// 	center:    input center of light
// 	color:     input color of light
// 	intensity: input intensity of light

sPointLight initLight(in vec3 pos, in vec3 color, in float intensity)
{
	sPointLight temp;
	
	temp.pos = pos;
	temp.color = color;
	temp.intensity = intensity;
	
	return temp;
}

void main()
{
	// Variable declarations
	// Constant to determine number of lights
	const int NUM_LIGHTS = 3;
	
	sPointLight lights[NUM_LIGHTS];

	mat4 modelViewMatrix = uViewMatrix * uModelMatrix;

	vec4 clipPos = uViewProjMatrix * uModelMatrix * aPosition, tPosition;
	
	vec3 tNormal, lightVector, tempVec, attenCoeff, diffuseIntensity, reflectedLight, 
		 viewVector, finalColor;
	
	float diffuseCoeff, specularCoeff;
	
	// Calculate position and normal in view space
	tPosition = modelViewMatrix * aPosition;
	tNormal = mat3(modelViewMatrix) * aNormal;
	
	// Main lighting calculation loop
	for (int i = 0; i < NUM_LIGHTS; i++)
	{
		// Initialize light at (i + 1, i + 1) in world space, white light, 1.5 intensity
		lights[i] = initLight(mat3(uModelMatrix) * vec3(i + 1, i + 1, 1), vec3(1), 1.5);
		
		// Calculate light vector
		lightVector = normalize(lights[i].pos);
	
		// Calculate diffuse coefficient
		diffuseCoeff = max(0.0, dot(aNormal, lightVector));
		diffuseCoeff = max(0.0, dot(tNormal, lightVector));
		
		// Calculate attenuation, use tempVec for efficiency
		tempVec = lights[i].pos / lights[i].intensity;
		attenCoeff = 1.0 / (tempVec + tempVec * tempVec);
		
		// Final diffuse intensity
		diffuseIntensity = diffuseCoeff * attenCoeff;
		
		// Calculate reflected light
		reflectedLight = reflect(-lightVector, aNormal);
		reflectedLight = reflect(-lightVector, tNormal);
		
		viewVector = normalize(lights[i].pos - vec3(aPosition));
		viewVector = normalize(lights[i].pos - vec3(tPosition));
		
		// Calculate specular coefficient
		specularCoeff = dot(reflectedLight, viewVector) * 0.5 + 0.5;

		// I don't know what the highlight exponent is, 
		// I just kept raising it until it looked good
		for (int j = 0; j < 10; j++)
		{
			specularCoeff *= specularCoeff;		
		}
		
		// Calculate final color for this light
		finalColor += diffuseIntensity * aNormal + specularCoeff * vec3(1) * lights[i].color;
	}
	
	// Factor in global light
	finalColor += 0.15 * vec3(1);
	
	// Set varyings
	vColor = vec4(finalColor, 0);

	vPosition = aPosition;
	//vPosition = tPosition;

	vNormal = aNormal;
	//vNormal = tNormal;
	
	// Set position in clip space
	gl_Position = clipPos;
}
